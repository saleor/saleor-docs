  availableCollectionPoints: [Warehouse!]!

  isPaid: Boolean!

  paymentStatusDisplay: String!
  authorizeStatus: OrderAuthorizeStatusEnum!
  chargeStatus: OrderChargeStatusEnum!

  languageCode: String!

  discount: Money
  discountName: String
  translatedDiscountName: String

}
