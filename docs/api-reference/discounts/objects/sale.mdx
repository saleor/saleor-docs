Sales allow creating discounts for categories, collections or products and are visible to all the customers.

DEPRECATED: this type will be removed. Use `Promotion` type instead.

```graphql
type Sale implements Node, ObjectWithMetadata {
  id: ID!
  privateMetadata: [MetadataItem!]!
  privateMetafield(
    key: String!
  ): String
  privateMetafields(
    keys: [String!]
  ): Metadata
  metadata: [MetadataItem!]!
  metafield(
    key: String!
  ): String
  metafields(
    keys: [String!]
  ): Metadata
  name: String!
  type: SaleType!
  startDate: DateTime!
  endDate: DateTime
  created: DateTime!
  updatedAt: DateTime!
  categories(
    before: String
    after: String
    first: Int
    last: Int
  ): CategoryCountableConnection
  collections(
    before: String
    after: String
    first: Int
    last: Int
  ): CollectionCountableConnection
  products(
    before: String
    after: String
    first: Int
    last: Int
  ): ProductCountableConnection
  variants(
    before: String
    after: String
    first: Int
    last: Int
  ): ProductVariantCountableConnection
  translation(
    languageCode: LanguageCodeEnum!
  ): SaleTranslation
  channelListings: [SaleChannelListing!]
  discountValue: Float
  currency: String
}
```
