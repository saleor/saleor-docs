import { useState } from "react";

export const Details = ({
  dataOpen,
  dataClose,
  children,
  startOpen = false,
}) => {
  const [open, setOpen] = useState(startOpen);
  return (
    <details
      {...(open ? { open: true } : {})}
      className="details"
      style={{
        border: "none",
        boxShadow: "none",
        background: "var(--ifm-background-color)",
      }}
    >
      <summary
        onClick={(e) => {
          e.preventDefault();
          setOpen((open) => !open);
        }}
        style={{ listStyle: "none" }}
      >
        {open ? dataOpen : dataClose}
      </summary>
      {open && children}
    </details>
  );
};

\_Entity union as defined by Federation spec.

```graphql
union _Entity =
    App
  | Address
  | User
  | Group
  | ProductVariant
  | Product
  | ProductType
  | ProductMedia
  | Category
  | Collection
  | Order
  | PageType
```

### Possible types

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>App</b></code>](../../../api-reference/apps/objects/app) <Badge class="secondary" text="object"/>

> Represents app data.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>Address</b></code>](../../../api-reference/users/objects/address) <Badge class="secondary" text="object"/>

> Represents user address data.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>User</b></code>](../../../api-reference/users/objects/user) <Badge class="secondary" text="object"/>

> Represents user data.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>Group</b></code>](../../../api-reference/users/objects/group) <Badge class="secondary" text="object"/>

> Represents permission group data.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>ProductVariant</b></code>](../../../api-reference/products/objects/product-variant) <Badge class="secondary" text="object"/>

> Represents a version of a product such as different size or color.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>Product</b></code>](../../../api-reference/products/objects/product) <Badge class="secondary" text="object"/>

> Represents an individual item for sale in the storefront.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>ProductType</b></code>](../../../api-reference/products/objects/product-type) <Badge class="secondary" text="object"/>

> Represents a type of product. It defines what attributes are available to products of this type.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>ProductMedia</b></code>](../../../api-reference/products/objects/product-media) <Badge class="secondary" text="object"/>

> Represents a product media.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>Category</b></code>](../../../api-reference/products/objects/category) <Badge class="secondary" text="object"/>

> Represents a single category of products. Categories allow to organize products in a tree-hierarchies which can be used for navigation in the storefront.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>Collection</b></code>](../../../api-reference/products/objects/collection) <Badge class="secondary" text="object"/>

> Represents a collection of products.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>Order</b></code>](../../../api-reference/orders/objects/order) <Badge class="secondary" text="object"/>

> Represents an order in the shop.

#### [<code style={{ fontWeight: 'normal' }}>\_Entity.<b>PageType</b></code>](../../../api-reference/pages/objects/page-type) <Badge class="secondary" text="object"/>

> Represents a type of page. It defines what attributes are available to pages of this type.
