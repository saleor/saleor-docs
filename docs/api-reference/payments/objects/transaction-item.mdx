Represents a payment transaction.

```graphql
type TransactionItem implements Node, ObjectWithMetadata {
  id: ID!
  privateMetadata: [MetadataItem!]!
  privateMetafield(
    key: String!
  ): String
  privateMetafields(
    keys: [String!]
  ): Metadata
  metadata: [MetadataItem!]!
  metafield(
    key: String!
  ): String
  metafields(
    keys: [String!]
  ): Metadata
  token: UUID!
  createdAt: DateTime!
  modifiedAt: DateTime!
  actions: [TransactionActionEnum!]!
  authorizedAmount: Money!
  authorizePendingAmount: Money!
  refundedAmount: Money!
  refundPendingAmount: Money!
  canceledAmount: Money!
  cancelPendingAmount: Money!
  chargedAmount: Money!
  chargePendingAmount: Money!
  name: String!
  message: String!
  pspReference: String!
  order: Order
  checkout: Checkout
  events: [TransactionEvent!]!
  createdBy: UserOrApp
  externalUrl: String!
  paymentMethodDetails: PaymentMethodDetails
  reason: String
  reasonReference: Page
}
```
