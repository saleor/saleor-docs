type Payment implements Node, ObjectWithMetadata {
  id: ID!

    key: String!

  privateMetafields(
    keys: [String!]
  ): Metadata
  metadata: [MetadataItem!]!
  metafield(
    key: String!

  metafields(
    keys: [String!]
  ): Metadata

  created: DateTime!

  token: String!

  order: Order
  paymentMethodType: String!

}
