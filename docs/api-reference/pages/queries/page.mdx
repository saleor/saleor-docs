import { useState } from "react";

export const Details = ({
  dataOpen,
  dataClose,
  children,
  startOpen = false,
}) => {
  const [open, setOpen] = useState(startOpen);
  return (
    <details
      {...(open ? { open: true } : {})}
      className="details"
      style={{
        border: "none",
        boxShadow: "none",
        background: "var(--ifm-background-color)",
      }}
    >
      <summary
        onClick={(e) => {
          e.preventDefault();
          setOpen((open) => !open);
        }}
        style={{ listStyle: "none" }}
      >
        {open ? dataOpen : dataClose}
      </summary>
      {open && children}
    </details>
  );
};

Look up a page by ID or slug.

```graphql
page(
  id: ID
  slug: String
): Page
```

### Arguments

#### [<code style={{ fontWeight: 'normal' }}>page.<b>id</b></code>](#)<Bullet />[`ID`](../../../api-reference/miscellaneous/scalars/id) <Badge class="secondary" text="scalar"/>

> ID of the page.

#### [<code style={{ fontWeight: 'normal' }}>page.<b>slug</b></code>](#)<Bullet />[`String`](../../../api-reference/miscellaneous/scalars/string) <Badge class="secondary" text="scalar"/>

> The slug of the page.

### Type

#### [`Page`](../../../api-reference/pages/objects/page) <Badge class="secondary" text="object"/>

> A static page that can be manually added by a shop operator through the dashboard.
