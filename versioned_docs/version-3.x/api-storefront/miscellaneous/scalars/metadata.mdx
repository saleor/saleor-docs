import { useState } from "react";

export const Details = ({
  dataOpen,
  dataClose,
  children,
  startOpen = false,
}) => {
  const [open, setOpen] = useState(startOpen);
  return (
    <details
      {...(open ? { open: true } : {})}
      className="details"
      style={{
        border: "none",
        boxShadow: "none",
        background: "var(--ifm-background-color)",
      }}
    >
      <summary
        onClick={(e) => {
          e.preventDefault();
          setOpen((open) => !open);
        }}
        style={{ listStyle: "none" }}
      >
        {open ? dataOpen : dataClose}
      </summary>
      {open && children}
    </details>
  );
};

Metadata is a map of key-value pairs, both keys and values are `String`.

Example:

```
&#x007B;
    "key1": "value1",
    "key2": "value2"
&#x007D;
```

```graphql
scalar Metadata
```
